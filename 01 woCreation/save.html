function onKeyDownHandler(event)
{





var evt = event;
if ( (evt == null) || (evt == undefined) )
{
evt = window.event;
}
if(!document.all)
{
if(evt.keyCode==17){
window.nonIECtrlDown = true;
return;
}
else if(evt.keyCode==18 ){
window.nonIEAltDown=true;
return;
}
}
if (isCtrlDown(evt))
{
if (isShiftDown(evt))
{
if (evt.keyCode == 75) // Ctrl+Shift+K: Summary of all the hotkeys provided by the system
{
loadHotKeySummaryWindow();
}
}
}
else if (isAltDown(evt))
{
if (evt.keyCode == 191) // Alt+/: Set focus on the fast path in the menu explorer
{
setFocusOnFastPath();
return false;
}
else if (evt.keyCode == 78) // Alt+N: Opens a new window
{
launchNewWindow();
return false;
}
}
if(evt.keyCode == 119) // F8 key to cycle the open application list
{
return processOpenApplicationList(evt);
}
if(evt.keyCode == 13) // capture Enter
{
if( (self.f8KeyPressed) && (OCLListFactory.getInstance()!= null))
{
self.f8KeyPressed = false; // reset the flag
}
return true;
}
if (isAltDown(evt))
{
if (evt.keyCode == 77) // Alt+M: Maximize/Minimize Left Nav
{
teMenuToggle();
return false;
}
}
return true;
}

var MenuEventQueue = new function()
{
this.SECONDS_TO_WAIT = 30;
this._events = new Array();
this._waiting = false;
this._eventStartTime = -1;
this.addEvent = function(eventObj)
{
this._events.push(eventObj);
this.execute();
};
this.execute = function()
{
if (this._waiting == false)
{
this._executeFirstEvent();
}
else
{
currentTime = (new Date()).getTime();
var timeDelta = currentTime - this._eventStartTime;
var continueWaiting = true;
if ( (this._events.length > 0) &&
(timeDelta >= 1000 * this.SECONDS_TO_WAIT) )
{
continueWaiting = confirm("The server is taking an excessive time to process your request. Click \"OK\" to continue waiting.");
if (continueWaiting)
{
this._eventStartTime = (new Date()).getTime();
}
}
if (continueWaiting)
{
setTimeout("MenuEventQueue.execute()",1000);
}
else
{
this._waiting = false;
this._executeFirstEvent();
}
}
};
this.eventComplete = function()
{
this._waiting = false;
setTimeout("MenuEventQueue.execute()",1);
}
this._executeFirstEvent = function()
{
if (this._events.length == 0)
{
this._waiting = false;
return;
}
var eventObj = this._events.shift();
this._waiting = true;
this._eventStartTime = (new Date()).getTime();
try
{
doActivateNodeWebGUI(eventObj.nodeId, eventObj.nodeId/*parent*/, eventObj.indentLevel);
var childDivId = "treechild" + eventObj.nodeId;
var childDiv = document.getElementById(childDivId);
var expanded = (childDiv.style.display == "block");
if (expanded && (eventObj.toggleMode == 'expand'))
{
return;
}
else if (expanded)
{
var theToggleIcon = document.getElementById(eventObj.nodeIconId);
theToggleIcon.src = "/jde/share/images/treecollapsed.gif";
childDiv.innerHTML = "";
childDiv.style.display = "none";
}
else if (eventObj.toggleMode == 'collapse')
{
return;
}
else
{
var theToggleIcon = document.getElementById(eventObj.nodeIconId);
theToggleIcon.src = "/jde/share/images/treeexpanded.gif";
childDiv.style.display = "block";
var loadingIndent = (eventObj.indentLevel + 1) * 12; // INDENT_SIZE
childDiv.innerHTML = "<table cellpadding=0 cellspacing=0><tr><td><img src='/jde/share/images/spacer.gif' width=" + loadingIndent + " height=3></td><td>&nbsp;Loading...</td></tr></table>";
}
document.getElementById("E1MTreeFormDiv").value = childDivId;
document.getElementById("E1MTreeFormNodeId").value = eventObj.nodeId;
document.getElementById("E1MTreeFormIndent").value = eventObj.indentLevel;
document.getElementById("E1MTreeFormRoleId").value = eventObj.roleId;
document.getElementById("E1MTreeFormCollapsed").value = expanded;
document.getElementById("E1MTreeForm").submit();
}
catch (problem)
{
this._waiting = false;
this.execute();
}
};
}
